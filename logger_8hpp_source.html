76 /**
77  * \brief Severity level
78  *
79  * This severity level can be specified for both console messages
80  * and file messages. It allows to disable messages (i.e., OFF),
81  * show only error messages (i.e. ERROR), show error and warning
82  * messages (i.e. WARN) or show all messages (i.e. DEBUG).
83  */
84 enum severity_level_t {
85  OFF = 0, //< No logging messages
86  ERROR = 1, //< Only errors
87  WARN = 2, //< Errors and warnings
88  DEBUG = 3 //< Show all messages
89 };
